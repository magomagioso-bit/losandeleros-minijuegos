/* NIVEL 4: SNAKE */ let snakeInterval; function iniciarSnake() { clearInterval(snakeInterval); const canvas = document.getElementById("snakeCanvas"); const ctx = canvas.getContext("2d"); const box = 20; let snake = [{x: 200, y: 200}]; let direction = null; // ⭐ NO SE MUEVE HASTA QUE PULSES let food = generarComida(); let gameOver = false; function generarComida() { return { x: Math.floor(Math.random()*20)*box, y: Math.floor(Math.random()*20)*box }; } // ⭐ Flechas + WASD document.onkeydown = (e) => { if (e.key === "ArrowUp" || e.key === "w") direction = "UP"; if (e.key === "ArrowDown" || e.key === "s") direction = "DOWN"; if (e.key === "ArrowLeft" || e.key === "a") direction = "LEFT"; if (e.key === "ArrowRight" || e.key === "d") direction = "RIGHT"; }; snakeInterval = setInterval(() => { if (gameOver) return; ctx.clearRect(0,0,400,400); // ⭐ Si no has pulsado nada, NO SE MUEVE if (!direction) { // dibujar serpiente quieta ctx.fillStyle = "lime"; snake.forEach(s => ctx.fillRect(s.x, s.y, box, box)); ctx.fillStyle = "red"; ctx.fillRect(food.x, food.y, box, box); return; } let head = {...snake[0]}; if (direction === "UP") head.y -= box; if (direction === "DOWN") head.y += box; if (direction === "LEFT") head.x -= box; if (direction === "RIGHT") head.x += box; // choque con paredes if (head.x < 0 || head.x >= 400 || head.y < 0 || head.y >= 400) { gameOver = true; document.getElementById("msg4").textContent = "¡Has perdido!"; setTimeout(() => volverMenu(), 1000); return; } snake.unshift(head); if (head.x === food.x && head.y === food.y) { ganarXP(5); food = generarComida(); } else { snake.pop(); } ctx.fillStyle = "red"; ctx.fillRect(food.x, food.y, box, box); ctx.fillStyle = "lime"; snake.forEach(s => ctx.fillRect(s.x, s.y, box, box)); }, 100); }
